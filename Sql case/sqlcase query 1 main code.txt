-- Brug master-databasen og fjern TEC-databasen, hvis den eksisterer

DROP DATABASE IF EXISTS TEC;
GO
CREATE DATABASE TEC;
GO
USE[TEC]

-- Opret tabel: elev, hvis den ikke eksisterer
IF NOT EXISTS (SELECT * FROM sys.tables WHERE name = 'elev')
CREATE TABLE elev (
    elev_id INT PRIMARY KEY,
    fornavn NVARCHAR(50), -- Ændret til NVARCHAR 
    efternavn NVARCHAR(50), -- Ændret til NVARCHAR
    adresse NVARCHAR(100), -- Ændret til NVARCHAR
    post_nr NVARCHAR(10), -- Ændret til NVARCHAR
    klasse_id NVARCHAR(10) -- Ændret til NVARCHAR
);
GO

-- Opret tabel: post_nr_by, hvis den ikke eksisterer
IF NOT EXISTS (SELECT * FROM sys.tables WHERE name = 'post_nr_by')
CREATE TABLE post_nr_by (
    post_nr NVARCHAR(10) PRIMARY KEY, -- Ændret til NVARCHAR
    by_navn NVARCHAR(50) -- Ændret til NVARCHAR
);
GO

-- Opret tabel: lærer, hvis den ikke eksisterer
IF NOT EXISTS (SELECT * FROM sys.tables WHERE name = 'lærer')
CREATE TABLE lærer (
    laerer_id INT PRIMARY KEY,
    fornavn NVARCHAR(50), -- Ændret til NVARCHAR
    efternavn NVARCHAR(50), -- Ændret til NVARCHAR
    adresse NVARCHAR(100), -- Ændret til NVARCHAR
    post_nr NVARCHAR(10), -- Ændret til NVARCHAR
    klasse_id NVARCHAR(10) -- Ændret til NVARCHAR
);
GO

-- Opret tabel: klasse, hvis den ikke eksisterer
IF NOT EXISTS (SELECT * FROM sys.tables WHERE name = 'klasse')
CREATE TABLE klasse (
    klasse_id NVARCHAR(10)  PRIMARY KEY -- Ændret til NVARCHAR
);
GO
